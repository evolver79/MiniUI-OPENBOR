
#### animationscript function main #####
# data/chars/Leo/Leo.txt
########################################
void main()
{
    int frame = getlocalvar("frame");
    int animhandle = getlocalvar("animhandle");
    if(animhandle==2)
    {
                                               //#  
  if(frame==1){                                       //#
    void self = getlocalvar("self");                  //#
                                                      //#
    changeentityproperty(self, "mp", 0);              //#
                                                      //#
  }                                                   //#
        return;
    }
    if(animhandle==38)
    {
        if(frame==2)
        {
            mpcost(25);
        }
        return;
    }
    if(animhandle==39)
    {
        if(frame==2)
        {
            mpcost(25);
        }
        return;
    }

}
